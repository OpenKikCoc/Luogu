## 题解



```c++
#include <bits/stdc++.h>
using namespace std;

const int N = 110;

int r, c, k;
char g[N][N];
int u[N][N], l[N][N];

int main() {
    cin >> r >> c >> k;
    
    for (int i = 1; i <= r; ++ i )
        cin >> (g[i] + 1);
    
    for (int i = 1; i <= r; ++ i )
        for (int j = 1; j <= c; ++ j )
            if (g[i][j] == '#')
                u[i][j] = 0;
            else
                u[i][j] = u[i - 1][j] + 1;
    for (int j = 1; j <= c; ++ j )
        for (int i = 1; i <= r; ++ i )
            if (g[i][j] == '#')
                l[i][j] = 0;
            else
                l[i][j] = l[i][j - 1] + 1;
    
    int res = 0;
    for (int i = 1; i <= r; ++ i )
        for (int j = 1; j <= c; ++ j )
            if (k > 1) {
                if (u[i][j] >= k)
                    res ++ ;
                if (l[i][j] >= k)
                    res ++ ;
            } else  if (k == 1 && (u[i][j] || l[i][j]))
                res ++ ;
        
    cout << res << endl;
    
    return 0;
}
```
